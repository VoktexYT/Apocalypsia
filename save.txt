       if (!this.actual_loader.isFinishedLoading() || !this.mesh) return;

        this.mesh.position.copy(object.player.camera.position);
        this.mesh.quaternion.copy(object.player.camera.quaternion);

        const settings = this.has_pistol ? this.PISTOL : this.RIFLE;
        const translate = this.is_shooting_position ? settings.shooting_position_translate : settings.normal_position_translate;
        const radians = this.is_shooting_position ? settings.shooting_position_radian : settings.normal_position_radian;

        this.mesh.translateX(translate[0]);
        this.mesh.translateY(translate[1] + Math.sin(this.movement) / (this.has_pistol ? 100 : 50));
        this.mesh.translateZ(translate[2]);
        this.mesh.rotateX(radians[0]);
        this.mesh.rotateY(radians[1]);
        this.mesh.rotateZ(radians[2]);

        if (this.fire_backward) {
            this.mesh.translateY(this.backward_intensity);
        }

        this.movement += 0.1;
